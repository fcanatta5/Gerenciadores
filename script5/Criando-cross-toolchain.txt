# Conceito do profile bootstrap,criar um profile chamado bootstrap cujo rootfs fica em:

/opt/adm/profiles/bootstrap/rootfs

# Dentro dele, vamos instalar o toolchain inicial em:

/opt/adm/profiles/bootstrap/rootfs/tools

# Criar o profile

sudo ./adm.sh init                 # inicializa estrutura em /opt/adm
./adm.sh profile set bootstrap     # cria/seleciona profile bootstrap
./adm.sh rootfs                    # só pra conferir o rootfs atual

# Criar o diretório das receitas

sudo mkdir -p /opt/adm/recipes
sudo chown -R "$(id -u)":"$(id -g)" /opt/adm/recipes
