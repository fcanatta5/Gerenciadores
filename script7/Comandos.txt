# Inicializar 
sudo /usr/adm/bin/adm init

# Escolher profile target,build+install
sudo /usr/adm/bin/adm build   --profile glibc.default --target x86_64-linux-gnu base/busybox-1.36.1
sudo /usr/adm/bin/adm install --profile glibc.default --target x86_64-linux-gnu base/busybox-1.36.1

# Buscar e ver info
sudo /usr/adm/bin/adm search busybox
sudo /usr/adm/bin/adm info base/busybox-1.36.1

# Órfãos 
sudo /usr/adm/bin/adm orphans --profile glibc.default --target x86_64-linux-gnu
sudo /usr/adm/bin/adm orphans --profile glibc.default --target x86_64-linux-gnu --remove

# Check,repara,reinstala programas do Word 
sudo /usr/adm/bin/adm check --profile glibc.default --target x86_64-linux-gnu
sudo /usr/adm/bin/adm check --profile glibc.default --target x86_64-linux-gnu --fix

# Construir com x86_64
sudo adm build \
  --profile glibc.default \
  --target x86_64-linux-gnu \
  toolchain/binutils-2.45.1-pass1

sudo adm install \
  --profile glibc.default \
  --target x86_64-linux-gnu \
  toolchain/binutils-2.45.1-pass1

# Cross-toolchain aarch64
sudo adm build \
  --profile glibc.default \
  --target aarch64-linux-gnu \
  toolchain/binutils-2.45.1-pass1

# Construir e instalar
sudo adm build --profile glibc.default --target aarch64-linux-gnu toolchain/gcc-15.2.0-pass1
sudo adm install --profile glibc.default --target aarch64-linux-gnu toolchain/gcc-15.2.0-pass1

# Construir com glibc
sudo adm build   --profile glibc.default --target aarch64-linux-gnu base/gcc-15.2.0
sudo adm install --profile glibc.default --target aarch64-linux-gnu base/gcc-15.2.0

# Construir com musl
sudo adm build   --profile musl.default --target aarch64-linux-musl base/gcc-15.2.0
sudo adm install --profile musl.default --target aarch64-linux-musl base/gcc-15.2.0
